35def article_create_view(request): 

36 """ 

37 Create article with user and images relationship 

38 """ 

39 

40 image_form_set = modelformset_factory(Image, form=ImageUploadForm, extra=5) 

41 equipment_form_set = modelformset_factory(Equipment, form=EquipmentForm, extra=15) 

42 # 'extra' means the number of photos that you can upload ^ 

43 if request.method == 'POST': 

44 

45 post_form = ArticleCreationForm(request.POST) 

46 formset = image_form_set(request.POST, request.FILES, queryset=Image.objects.none()) 

47 formset_equipment = equipment_form_set(request.POST, request.FILES, queryset=Equipment.objects.none()) 

48 

49 if post_form.is_valid() and formset.is_valid() and formset_equipment.is_valid(): 

50 form_post = post_form.save(commit=False) 

51 form_post.user = request.user 

52 article_instance = post_form.save() 

53 image_instances = formset.save() 

54 equipment_instances = formset_equipment.save() 

55 

56 # many to many relationship of article with author_bibliography 

57 print('request.user.id_account ', request.user.id_account) 

58 qs_username = Author.objects.get(account_author_id=request.user.id_account) 

59 print('qs_username', qs_username) 

60 print('article_instance', article_instance) 

61 qs_username.bibliography.add(article_instance) 

62 # many to many relationship of all images with article_list_image 

63 for image_instance in image_instances: 

64 article_instance.list_image.add(image_instance) 

65 

66 # many to many relationship of article with equipment for all equipment list 

67 for equipment in equipment_instances: 

68 article_instance.list_equipement.add(equipment) 

69 

70 messages.success(request, 

71 "Yeeew, check it out on the home page!") 

72 

73 return HttpResponseRedirect("/") 

74 else: 

75 print(post_form.errors, formset.errors) 

76 else: 

77 post_form = ArticleCreationForm() 

78 formset = image_form_set(queryset=Image.objects.none()) 

79 formset_equipment = equipment_form_set(queryset=Equipment.objects.none()) 

80 

81 return render(request, 'ArticlesApp/article_create.html', {'postForm': post_form, 'formset': formset, 

82 'formset_equipment': formset_equipment})
